	id = shroud.3
	title = BLANK_STRING
	diplomatic_title = shroud.3.diplo_title
	desc = BLANK_STRING

	is_triggered_only = yes
	diplomatic = yes
	force_open = yes

	picture_event_data = {
		portrait = shroud1
		room = shroud_room
	}

	option = {
		name = BLANK_STRING
	}

	show_sound = {
		trigger = { is_robot_empire = yes }
		sound = event_psionic_robot
	}
	show_sound = {
		trigger = { is_robot_empire = no }
		sound = event_psionic
	}
}

# The Spireless Tower - The Shining Path
country_event = {
	id = shroud.25025
	base = shroud.3

	title = shroud.25025.title
	desc_clear = yes
	desc = shroud.25025.desc

	trigger = {
		NOR = {
			has_country_flag = shroud_25025_happened
			has_country_flag = forging_our_own_path
		}
		has_patron_relation = {
			patron = mmp_the_spireless_tower
			contact = completed
		}
		OR = {
			is_in_domain = the_eater_of_worlds
			is_in_domain = the_instrument_of_desire
		}
	}

	immediate = {
		set_country_flag = shroud_25025_happened
	}

	option_clear = yes
	option = {
		name = shroud.25025.a
		custom_tooltip = shroud.25025.a.tt
	}
}
